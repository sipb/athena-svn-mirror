enum ChunkTypes

typedef enum {
CHUNKER_RESET,
CHUNKER_NOCHUNKS,
CHUNKER_KEYS,
CHUNKER_ROWS,
CHUNKER_COLUMNS,
CHUNKER_RECURSIVE,
CHUNKER_2GROUPS,
CHUNKER_3GROUPS,
CHUNKER_4GROUPS,
CHUNKER_5GROUPS,
CHUNKER_6GROUPS
} ChunkTypes;

enum ChunkOrder

typedef enum {
CHUNKER_TOPTOBOTTOM_LEFTTORIGHT,
CHUNKER_TOPTOBOTTOM_RIGHTTOLEFT,
CHUNKER_BOTTOMTOTOP_LEFTTORIGHT,
CHUNKER_BOTTOMTOTOP_RIGHTTOLEFT,
CHUNKER_LEFTTORIGHT_TOPTOBOTTOM,
CHUNKER_LEFTTORIGHT_BOTTOMTOTOP,
CHUNKER_RIGHTTOLEFT_TOPTOBOTTOM,
CHUNKER_RIGHTTOLEFT_BOTTOMTOTOP
} ChunkOrder;

gok_chunker_chunk_group ()

GokChunk*   gok_chunker_chunk_group         (GokChunk *pChunk,
                                             gint NumberGroups,
                                             ChunkOrder Order,
                                             gboolean bCanReorder);

Breaks the given list of chunks into 2 or more groups.

pChunk : Pointer to the list of chunks that will be broken into groups.
NumberGroups : Number of groups the chunks should be broken into.
Order : The order in which the chunks should be created. (see enum ChunkOrder)
bCanReorder : TRUE if the chunks can be reordered.
Returns : A pointer to the first chunk in the list of chunks. Returns NULL if the given chunk can't be divided into chunks.
